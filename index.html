CREATE TABLE IF NOT EXISTS lucky_draw_config (
  id          int primary key default 1,
  enabled     boolean default true,
  title       text default '新人专享福利',
  subtitle    text default '限时领取 · 今日有效',
  daily_limit int default 1,     -- 每日基础次数
  share_extra int default 2,     -- 分享最多额外次数
  updated_at  timestamptz default now()
);

CREATE TABLE IF NOT EXISTS lucky_draw_prizes (
  id          serial primary key,
  name        text not null,
  emoji       text default '🎁',
  type        text default 'coupon',
  value       numeric default 0,
  min_order   numeric default 0,
  probability int default 10,
  color       text default '#FFF0F0',
  enabled     boolean default true,
  sort_order  int default 0
);

系统设置
数据库字段说明（goods 表）
字段名类型说明
idint8主键自增
nametext商品名称
pricenumeric售价（小程序显示此字段）
original_pricenumeric划线原价（可空）
unittext计量单位，如 斤/个/盒
category_idint21蔬菜 2水果 3肉禽 4海鲜 5蛋奶 6粮油 7豆制品 8日用品
stockint4库存数量
stock_warnint4预警阈值
tagstext卖点标签，逗号分隔
detailtext商品详情描述
is_hotbool加入今日特惠
is_newbool加入新品上架
statusint20下架 1上架
sold_countint4累计销售数
created_attimestamptz创建时间
⚠️ 旧表有 group_price / group_size 字段，新版统一用 price。 建议在 Supabase 执行：
ALTER TABLE goods ADD COLUMN IF NOT EXISTS price numeric;
UPDATE goods SET price = group_price WHERE price IS NULL;
ALTER TABLE goods ADD COLUMN IF NOT EXISTS unit text DEFAULT '件';
